class Solution {
    
    public List<Integer> leaders(int[] nums) {

        List<Integer> list=new ArrayList<>();
        for(int i=0;i<nums.length;i++){
            
            boolean leader=true; 
        for(int j=i+1;j<nums.length;j++){

                if(nums[j]>nums[i]){
                    leader=false;
                }

                }
                if(leader){
                    list.add(nums[i]);
                }
            }
        
        return list;

    }
}
