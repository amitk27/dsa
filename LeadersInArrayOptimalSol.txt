class Solution {
    
    public List<Integer> leaders(int[] nums) {

        List<Integer> list=new ArrayList<>();
        int max=-999999999;
        for(int i=nums.length-1;i>=0;i--){

            if(nums[i]>max){
            list.add(nums[i]);
            }
            max=Math.max(max,nums[i]);

        }
Collections.reverse(list);
        return list;
       

    }
}
