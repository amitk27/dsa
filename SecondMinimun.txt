class Solution {
    public int secondLargestElement(int[] nums) {

     int min = nums[0];
     int secondMinimun= Integer.MAX_VALUE;
       for(int i =0;i<nums.length;i++){
        if(nums[i]<min){
            secondMinimun=min;
            min=nums[i];


        }else if(nums[i]>min && nums[i]<secondMinimun){
                  secondMinimun=nums[i];
        }

       }
        return secondMinimun;
  
    }
}
